Precast Icons - HTML Example

Using Precast Icons in vanilla HTML/JavaScript applications

CDN Usage

Include the library via CDN and use icons directly:

<!-- Include Precast Icons from CDN -->
<script src="https://unpkg.com/@precast/icons/dist/icons.min.js"></script>

<!-- Use icons in your HTML -->
<div id="icon-container"></div>

<script>
  // Access icons from the global PrecastIcons object
  const { icons } = PrecastIcons;
  
  // Render an icon
  const container = document.getElementById('icon-container');
  container.innerHTML = renderIcon(icons.home, { size: 24, color: '#3b82f6' });
</script>

    Module Usage (ES6)

    import { icons } from '@precast/icons';

// Function to render an icon
function renderIcon(icon, options = {}) {
  const {
    size = 24,
    color = 'currentColor',
    strokeWidth = 2,
    fill = 'none',
    stroke = color
  } = options;

  const svgContent = icon.content
    .map(node => renderNode(node))
    .join('');

  return `<svg 
    width="${size}" 
    height="${size}" 
    viewBox="${icon.viewBox || '0 0 24 24'}"
    fill="${fill}"
    stroke="${stroke}"
    stroke-width="${strokeWidth}"
    stroke-linecap="round"
    stroke-linejoin="round"
  >${svgContent}</svg>`;
}

function renderNode(node) {
  const attrs = Object.entries(node.attrs || {})
    .map(([key, value]) => `${key}="${value}"`)
    .join(' ');
  
  if (node.children && node.children.length > 0) {
    const children = node.children.map(renderNode).join('');
    return `<${node.tag} ${attrs}>${children}</${node.tag}>`;
  }
  
  return `<${node.tag} ${attrs} />`;
}

// Use the icon
document.getElementById('my-icon').innerHTML = renderIcon(icons.home, {
  size: 32,
  color: '#3b82f6'
});
